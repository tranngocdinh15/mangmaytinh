Linear SearchC
int LinearSearch(int a[], int n, int x) {
    for (int i = 0; i < n; i++)
        if (a[i] == x) return i;  // tìm thấy
    return -1;  // không tìm thấy
}
Binary SearchC
int BinarySearch(int a[], int n, int x) {
    int left = 0, right = n - 1;
    while (left <= right) {              // điều kiện: <=
        int mid = (left + right) / 2;
        if      (a[mid] == x) return mid;
        else if (a[mid] <  x) left  = mid + 1;  // tìm phải
        else                  right = mid - 1;  // tìm trái
    }
    return -1;
}
Interchange SortC
void InterchangeSort(int a[], int n) {
    for (int i = 0; i < n-1; i++)           // xuất phát từ ĐẦU
        for (int j = i+1; j < n; j++)
            if (a[i] > a[j]) {               // nghịch thế → đổi chỗ
                int t=a[i]; a[i]=a[j]; a[j]=t;
            }
}
Bubble SortC
void BubbleSort(int a[], int n) {
    for (int i = 0; i < n-1; i++)
        for (int j = n-1; j > i; j--)  // xuất phát từ CUỐI
            if (a[j] < a[j-1]) {          // cặp kề nhau
                int t=a[j]; a[j]=a[j-1]; a[j-1]=t;
            }
}
Selection SortC
void SelectionSort(int a[], int n) {
    for (int i = 0; i < n-1; i++) {
        int mn = i;
        for (int j = i+1; j < n; j++)
            if (a[j] < a[mn]) mn = j;    // chỉ lưu vị trí min
        if (mn != i) { int t=a[i]; a[i]=a[mn]; a[mn]=t; } // swap 1 lần
    }
}
Insertion SortC
void InsertionSort(int a[], int n) {
    for (int i = 1; i < n; i++) {   // bắt đầu từ i=1!
        int key = a[i], j = i-1;
        while (j >= 0 && a[j] > key) { a[j+1]=a[j]; j--; }
        a[j+1] = key;               // chèn key vào đúng chỗ
    }
}
Quick SortC
void QuickSort(int a[], int L, int R) {
    int i=L, j=R, x=a[(L+R)/2];  // pivot = phần tử giữa
    while (i <= j) {
        while (a[i] < x) i++;
        while (a[j] > x) j--;
        if (i <= j) { int t=a[i]; a[i]=a[j]; a[j]=t; i++; j--; }
    }
    if (L < j) QuickSort(a, L, j);
    if (i < R) QuickSort(a, i, R);
}
Merge SortC
void merge(int a[], int l, int m, int r) {
    int n1=m-l+1, n2=r-m, L[n1], R[n2];
    for(int i=0;i<n1;i++) L[i]=a[l+i];
    for(int j=0;j<n2;j++) R[j]=a[m+1+j];
    int i=0,j=0,k=l;
    while(i<n1 && j<n2) a[k++]=(L[i]<=R[j])?L[i++]:R[j++];
    while(i<n1) a[k++]=L[i++];
    while(j<n2) a[k++]=R[j++];
}
void MergeSort(int a[], int l, int r) {
    if(l < r) { int m=(l+r)/2; MergeSort(a,l,m); MergeSort(a,m+1,r); merge(a,l,m,r); }
}
📊 Big-O nhanh

O(1) < O(log n) < O(n) < O(n log n) < O(n²)

🟢 O(1)

  • Truy cập a[i]
  • Binary Search best case

🔵 O(log n)

  • Binary Search avg/worst
  • n=1024 → chỉ 10 bước

🟡 O(n)

  • Linear Search avg/worst
  • Insertion Sort best case!

🟠 O(n log n)

  • Quick Sort best/avg
  • Merge Sort mọi TH

🔴 O(n²)

  • Interchange, Bubble, Selection, Insertion
  • Quick Sort worst case

📌 Công thức so sánh

  • Linear TB: (n+1)/2
  • Interchange/Bubble/Selection: n(n-1)/2
  • Insertion best: n-1

❗ Bẫy 1

  • Interchange j từ i+1, Bubble j từ n-1
  • Interchange unstable, Bubble stable

❗ Bẫy 2

  • Insertion: best O(n) khi mảng đã sắp
  • Duy nhất O(n²) có best O(n)

❗ Bẫy 3

  • Quick Sort worst O(n²)
  • Khi mảng đã sắp + pivot đầu/cuối

❗ Bẫy 4

  • Binary Search bắt buộc mảng có thứ tự
  • Hay hỏi "điều kiện áp dụng"
