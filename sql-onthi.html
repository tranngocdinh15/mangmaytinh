SQL — Ôn Thi Trắc Nghiệm

Lý thuyết + 30 câu hỏi trắc nghiệm · Từ cơ bản đến nâng cao

1 · DDL — Định nghĩa dữ liệu

CREATE TABLE

Tạo bảng mới trong CSDL.

CREATE TABLE NhanVien (
  id   INT PRIMARY KEY,
  ten  VARCHAR(100),
  luong DECIMAL(10,2)
);

ALTER TABLE

Thêm / sửa / xoá cột hoặc ràng buộc.

ALTER TABLE NhanVien
  ADD email VARCHAR(200);

ALTER TABLE NhanVien
  DROP COLUMN email;

DROP / TRUNCATE

DROP xoá bảng. TRUNCATE xoá toàn bộ dữ liệu nhưng giữ cấu trúc.

DROP TABLE NhanVien;

TRUNCATE TABLE NhanVien;

Constraints

Ràng buộc dữ liệu: PK, FK, UNIQUE, NOT NULL, CHECK, DEFAULT.

id   INT PRIMARY KEY,
dep  INT REFERENCES PhongBan(id),
ten  VARCHAR(50) NOT NULL,
age  INT CHECK (age > 18)
2 · DML — Thao tác dữ liệu

INSERT

Chèn hàng mới vào bảng.

INSERT INTO NhanVien (id, ten, luong)
VALUES (1, 'Minh', 15000000);

-- Chèn từ SELECT
INSERT INTO Archive
SELECT * FROM NhanVien WHERE luong < 5e6;

UPDATE

Cập nhật dữ liệu. Nhớ thêm WHERE!

UPDATE NhanVien
SET   luong = luong * 1.1,
      email = 'new@co.vn'
WHERE id = 1;

DELETE

Xoá hàng khỏi bảng.

DELETE FROM NhanVien
WHERE luong < 3000000;

-- Xoá toàn bộ (cẩn thận!)
DELETE FROM NhanVien;

MERGE (UPSERT)

Kết hợp INSERT + UPDATE + DELETE trong 1 lệnh.

MERGE INTO Target AS T
USING Source AS S
  ON T.id = S.id
WHEN MATCHED THEN UPDATE ...
WHEN NOT MATCHED THEN INSERT ...;
3 · DQL — Truy vấn dữ liệu

SELECT cơ bản

Lấy dữ liệu từ bảng.

SELECT ten, luong
FROM   NhanVien
WHERE  luong > 10000000
ORDER BY luong DESC
LIMIT  5;

GROUP BY + HAVING

HAVING lọc nhóm (tương tự WHERE nhưng dùng sau GROUP BY).

SELECT   phong, AVG(luong) AS avg_luong
FROM     NhanVien
GROUP BY phong
HAVING   AVG(luong) > 8000000;

Aggregate Functions

Hàm tổng hợp thường dùng.

COUNT(*)          -- đếm tất cả
COUNT(DISTINCT x) -- đếm phân biệt
SUM(luong)
AVG(luong)
MIN(luong), MAX(luong)

Toán tử lọc

Các điều kiện trong WHERE.

BETWEEN 5e6 AND 15e6 IN ('IT', 'HR', 'Sales') LIKE 'Nguy%' -- % nhiều ký tự LIKE 'Minh_' -- _ 1 ký tự IS NULL / IS NOT NULL
4 · JOIN — Kết hợp bảng

INNER JOIN

Chỉ lấy hàng khớp ở cả hai bảng.

SELECT nv.ten, pb.ten_phong
FROM   NhanVien nv
INNER JOIN PhongBan pb
  ON nv.phong_id = pb.id;

LEFT / RIGHT JOIN

Giữ toàn bộ bảng trái (LEFT) hoặc phải (RIGHT), bên kia NULL nếu không khớp.

SELECT nv.ten, pb.ten_phong
FROM   NhanVien nv
LEFT JOIN PhongBan pb
  ON nv.phong_id = pb.id;

FULL OUTER JOIN

Giữ tất cả hàng từ cả hai bảng.

SELECT *
FROM   A
FULL OUTER JOIN B
  ON A.id = B.id;

SELF JOIN & CROSS JOIN

SELF JOIN: bảng join với chính nó. CROSS JOIN: tích Đề-các.

-- SELF JOIN: tìm quản lý
SELECT e.ten, m.ten AS quan_ly
FROM  NhanVien e
JOIN  NhanVien m ON e.mgr_id = m.id;
5 · Subquery & Set Operators

Subquery

Truy vấn lồng trong truy vấn khác.

SELECT ten FROM NhanVien
WHERE  luong = (
  SELECT MAX(luong) FROM NhanVien
);

EXISTS / NOT EXISTS

Kiểm tra sự tồn tại của kết quả subquery.

SELECT ten FROM PhongBan pb
WHERE EXISTS (
  SELECT 1 FROM NhanVien nv
  WHERE nv.phong_id = pb.id
);

UNION / INTERSECT / EXCEPT

Gộp, giao, hiệu tập kết quả.

SELECT ten FROM A
UNION          -- loại trùng
SELECT ten FROM B;

UNION ALL      -- giữ trùng
INTERSECT      -- giao
EXCEPT         -- hiệu

CTE (WITH)

Common Table Expression — đặt tên cho subquery.

WITH HighEarner AS (
  SELECT * FROM NhanVien
  WHERE  luong > 20e6
)
SELECT * FROM HighEarner
ORDER BY luong;
6 · Window Functions

ROW_NUMBER / RANK

Đánh số thứ tự trong nhóm.

SELECT ten, luong,
  ROW_NUMBER() OVER (
    PARTITION BY phong
    ORDER BY luong DESC
  ) AS hang
FROM NhanVien;

SUM / AVG OVER

Tính tổng/trung bình tích lũy hoặc theo cửa sổ.

SELECT ten,
  SUM(luong) OVER (
    ORDER BY id
    ROWS UNBOUNDED PRECEDING
  ) AS luy_ke
FROM NhanVien;

LAG / LEAD

Lấy giá trị hàng trước/sau.

SELECT ngay, doanh_thu,
  LAG(doanh_thu, 1) OVER (
    ORDER BY ngay
  ) AS hom_truoc
FROM DoanhThu;

NTILE / DENSE_RANK

NTILE chia đều thành N nhóm. DENSE_RANK không bỏ số khi trùng.

RANK()       -- 1,2,2,4
DENSE_RANK() -- 1,2,2,3
NTILE(4)    -- chia 4 tứ phân vị
7 · TCL — Điều khiển giao dịch

TRANSACTION

Nhóm lệnh thành đơn vị nguyên tử (ACID).

BEGIN TRANSACTION;
  UPDATE TaiKhoan SET so_du = so_du - 100 WHERE id=1;
  UPDATE TaiKhoan SET so_du = so_du + 100 WHERE id=2;
COMMIT;

ROLLBACK & SAVEPOINT

Hủy giao dịch hoặc quay về điểm lưu.

SAVEPOINT sp1;
UPDATE ...;
-- lỗi xảy ra
ROLLBACK TO sp1;
COMMIT;

INDEX

Tăng tốc truy vấn, chậm hơn khi ghi.

CREATE INDEX idx_ten
  ON NhanVien(ten);

CREATE UNIQUE INDEX idx_email
  ON NhanVien(email);

VIEW

Bảng ảo lưu truy vấn SELECT.

CREATE VIEW v_luong_cao AS
  SELECT ten, luong
  FROM   NhanVien
  WHERE  luong > 20e6;

SELECT * FROM v_luong_cao;
